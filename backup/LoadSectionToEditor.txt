procedure TfrmMain.LoadSectionToEditor(const ConfigPath, SectionName: string);
var
  FileExt: string;
  IniFile: TMemIniFile;
  JsonFile: TJSONObject;
  Keys: TStringList;
  i: Integer;
  Key, Value: string;
  FileContent: string;
  JsonProperty: TJSONPair;
  EncodingUsed: TEncoding;
  EncodingOptions: array of TEncoding;
  SuccessfullyLoaded: Boolean;
begin
  try
    FormHelper.AddDebugText(meoDebug, '===== 开始加载节内容 =====');
    
    // 清空当前编辑器内容
    FValueListEditor.Strings.Clear;
    
    // 保存当前配置路径
    FCurrentBasicConfigPath := ConfigPath;
    
    // 根据文件扩展名确定配置类型
    FileExt := LowerCase(ExtractFileExt(ConfigPath));
    
    FormHelper.AddDebugText(meoDebug, '加载文件: ' + ConfigPath);
    FormHelper.AddDebugText(meoDebug, '节名: ' + SectionName + ' (字节长度: ' + IntToStr(ByteLength(SectionName)) + ')');
    
    if FileExt = '.ini' then
    begin
      // 定义尝试的编码顺序
      EncodingOptions := [TEncoding.UTF8, TEncoding.Unicode, TEncoding.Default, TEncoding.ASCII];
      SuccessfullyLoaded := False;
      
      // 尝试使用不同的编码加载INI文件
      for EncodingUsed in EncodingOptions do
      begin
        if SuccessfullyLoaded then
          Break;
          
        FormHelper.AddDebugText(meoDebug, '尝试使用编码: ' + EncodingUsed.ClassName);
        
        try
          IniFile := TMemIniFile.Create(ConfigPath, EncodingUsed);
          try
            Keys := TStringList.Create;
            try
              // 添加节标题
              try
                var SectionHeader := '[' + SectionName + ']';
                FormHelper.AddDebugText(meoDebug, '正在添加节标题: ' + SectionHeader);
                
                FValueListEditor.InsertRow(SectionHeader, '', True);
                FormHelper.AddDebugText(meoDebug, '成功添加节标题');
                
                // 读取该节中的所有键
                IniFile.ReadSection(SectionName, Keys);
                
                FormHelper.AddDebugText(meoDebug, '节包含 ' + IntToStr(Keys.Count) + ' 个键');
                
                for i := 0 to Keys.Count - 1 do
                begin
                  Key := Keys[i];
                  Value := IniFile.ReadString(SectionName, Key, '');
                  
                  // 添加到值列表编辑器
                  try
                    var FullKey := SectionName + '.' + Key;
                    FormHelper.AddDebugText(meoDebug, '添加键值对: ' + FullKey + ' = ' + Value);
                    
                    FValueListEditor.Values[FullKey] := Value;
                  except
                    on E: Exception do
                    begin
                      FormHelper.AddDebugText(meoDebug, '添加键值对时出错: ' + E.Message);
                      // 继续尝试添加其他键值对
                    end;
                  end;
                end;
                
                SuccessfullyLoaded := True;
                FormHelper.AddDebugText(meoDebug, '使用 ' + EncodingUsed.ClassName + ' 编码成功加载节内容');
              except
                on E: Exception do
                begin
                  FormHelper.AddDebugText(meoDebug, '使用 ' + EncodingUsed.ClassName + ' 编码添加节标题时出错: ' + E.Message);
                  // 将在下一次循环中尝试其他编码
                end;
              end;
            finally
              Keys.Free;
            end;
          finally
            IniFile.Free;
          end;
        except
          on E: Exception do
          begin
            FormHelper.AddDebugText(meoDebug, '使用 ' + EncodingUsed.ClassName + ' 编码加载文件时出错: ' + E.Message);
            // 将在下一次循环中尝试其他编码
          end;
        end;
      end;
      
      if not SuccessfullyLoaded then
      begin
        FormHelper.AddDebugText(meoDebug, '所有编码尝试均失败，无法加载节内容');
        ShowMessage('无法加载配置节: ' + SectionName + '。可能包含不支持的字符编码。');
      end;
    end
    else if FileExt = '.json' then
    begin
      // 尝试使用不同的编码加载JSON文件
      EncodingOptions := [TEncoding.UTF8, TEncoding.Unicode, TEncoding.Default];
      SuccessfullyLoaded := False;
      
      for EncodingUsed in EncodingOptions do
      begin
        if SuccessfullyLoaded then
          Break;
          
        FormHelper.AddDebugText(meoDebug, '尝试使用编码: ' + EncodingUsed.ClassName + ' 加载JSON文件');
        
        try
          if FileExists(ConfigPath) then
          begin
            try
              // 使用指定编码读取文件
              FileContent := TFile.ReadAllText(ConfigPath, EncodingUsed);
              JsonFile := TJSONObject(TJSONObject.ParseJSONValue(FileContent));
              
              if JsonFile <> nil then
              begin
                try
                  // 添加JSON根节点标题
                  try
                    FValueListEditor.InsertRow('[JSON根节点]', '', True);
                    FormHelper.AddDebugText(meoDebug, '添加JSON根节点标题成功');
                  except
                    on E: Exception do
                    begin
                      FormHelper.AddDebugText(meoDebug, '添加JSON根节点标题时出错: ' + E.Message);
                      Continue; // 尝试下一种编码
                    end;
                  end;
                
                  // 遍历JSON对象的所有键值对
                  for i := 0 to JsonFile.Count - 1 do
                  begin
                    JsonProperty := JsonFile.Pairs[i];
                    Key := JsonProperty.JsonString.Value;
                    
                    // 将JSON值转换为字符串
                    if JsonProperty.JsonValue is TJSONString then
                      Value := TJSONString(JsonProperty.JsonValue).Value
                    else if JsonProperty.JsonValue is TJSONNumber then
                      Value := TJSONNumber(JsonProperty.JsonValue).ToString
                    else if JsonProperty.JsonValue is TJSONBool then
                      Value := BoolToStr(TJSONBool(JsonProperty.JsonValue).AsBoolean, True)
                    else if JsonProperty.JsonValue is TJSONObject then
                      Value := '[Object]'
                    else if JsonProperty.JsonValue is TJSONArray then
                      Value := '[Array]'
                    else
                      Value := JsonProperty.JsonValue.ToString;
                    
                    // 添加到值列表编辑器
                    try
                      var FullKey := 'JSON根节点.' + Key;
                      FormHelper.AddDebugText(meoDebug, '添加JSON键值对: ' + FullKey);
                      
                      FValueListEditor.Values[FullKey] := Value;
                    except
                      on E: Exception do
                      begin
                        FormHelper.AddDebugText(meoDebug, '添加JSON键值对时出错: ' + E.Message);
                        // 继续尝试添加其他键值对
                      end;
                    end;
                  end;
                  
                  SuccessfullyLoaded := True;
                  FormHelper.AddDebugText(meoDebug, '使用 ' + EncodingUsed.ClassName + ' 编码成功加载JSON内容');
                finally
                  JsonFile.Free;
                end;
              end;
            except
              on E: Exception do
              begin
                FormHelper.AddDebugText(meoDebug, '使用 ' + EncodingUsed.ClassName + ' 编码解析JSON失败: ' + E.Message);
                // 将在下一次循环中尝试其他编码
              end;
            end;
          end;
        except
          on E: Exception do
          begin
            FormHelper.AddDebugText(meoDebug, '使用 ' + EncodingUsed.ClassName + ' 编码加载JSON内容失败: ' + E.Message);
            // 将在下一次循环中尝试其他编码
          end;
        end;
      end;
      
      if not SuccessfullyLoaded then
      begin
        FormHelper.AddDebugText(meoDebug, '所有编码尝试均失败，无法加载JSON内容');
        ShowMessage('无法加载JSON配置。可能包含不支持的字符编码。');
      end;
    end;
    
    // 显示基础信息选项卡
    FBasicInfoTab.TabVisible := True;
    pcEditors.ActivePage := FBasicInfoTab;
    
    FormHelper.AddDebugText(meoDebug, '===== 节内容加载完成 =====');
  except
    on E: Exception do
    begin
      FormHelper.AddDebugText(meoDebug, '加载节时出现未处理的异常: ' + E.Message);
      ShowMessage('出现错误: ' + E.Message);
    end;
  end;
end;
